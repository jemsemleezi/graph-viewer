    function initGraph() {
        const container = document.getElementById('graph-canvas');
        const data = { nodes, edges };
        const options = {
            physics: { enabled: true, solver: 'forceAtlas2Based', forceAtlas2Based: { gravitationalConstant: -100, centralGravity: 0.005, springLength: 120, springConstant: 0.08, damping: 0.85 }, stabilization: { enabled: true, iterations: 100, updateInterval: 15 }, wind: { x: 0, y: 0 } },
            interaction: { hover: true, tooltipDelay: 200, dragView: true },
            edges: { arrows: 'to', smooth: { type: 'dynamic' } }
        };
        network = new vis.Network(container, data, options);
        network.on('click', (params) => {
            if (params.nodes.length > 0) showNodeDetail(params.nodes[0]);
            else if (params.edges.length > 0) showEdgeDetail(params.edges[0]);
            else hideDetailPanel();
        });
    }

    function hexToRgba(hex, alpha) {
        if (!hex || hex.startsWith('rgba')) return hex;
        let r = 0, g = 0, b = 0;
        if (hex.length === 4) { r = parseInt(hex[1] + hex[1], 16); g = parseInt(hex[2] + hex[2], 16); b = parseInt(hex[3] + hex[3], 16); }
        else if (hex.length === 7) { r = parseInt(hex.substring(1, 3), 16); g = parseInt(hex.substring(3, 5), 16); b = parseInt(hex.substring(5, 7), 16); }
        return `rgba(${r}, ${g}, ${b}, ${alpha})`;
    }

    async function loadGraph() {
        try {
            const res = await fetch('/api/graph'); const data = await res.json();
            fullGraphData = data;
            const formattedNodes = data.nodes.map(n => ({ id: n.id, label: n.name || n.id, title: `<b>${n.name}</b><br>Type: ${n.type}<br>${n.description || ''}`, color: { background: nodeColors[n.type] || '#6b7280', border: nodeColors[n.type] || '#6b7280', highlight: { background: '#60a5fa', border: '#60a5fa' } }, font: { color: '#ffffff' }, shape: 'dot', size: 15 + (n.pagerank || 0) * 50, _rawType: n.type }));
            const formattedEdges = data.edges.map(e => ({ id: e.id, from: e.from_id, to: e.to_id, label: e.type, arrows: 'to', width: 1.5, selectionWidth: 5, hoverWidth: 3, smooth: { type: 'dynamic' }, color: { color: edgeColors[e.type] || '#4b5563', highlight: '#3b82f6' }, font: { color: '#a2a5b9', strokeWidth: 0, size: 10 }, instruction: e.instruction, condition: e.condition, _rawType: e.type }));
            nodes.clear(); edges.clear(); nodes.add(formattedNodes); edges.add(formattedEdges);
            network.stabilize(100);
            filterLocalData();
            await loadStats();
        } catch (e) { showToast('加载数据失败: ' + e.message, 'error'); }
    }

    async function loadStats() {
        try {
            const [statsRes, sourceRes, dbRes] = await Promise.all([fetch('/api/stats'), fetch('/api/source'), fetch('/api/current-db')]);
            const stats = await statsRes.json(); const source = await sourceRes.json(); const db = await dbRes.json();
            
            let html = '';
            let sourceText = source.mode === 'json' ? 'JSON (内存模式)' : `SQLite (.../${db.path.replace(/\\/g, '/').split('/').slice(-2).join('/')})`;
            html += `<div class="stats-source" title="当前连接的数据源">🔗 ${sourceText}</div>`;
            
            html += `<div class="stats-row">
                        <div class="stats-item"><div class="stats-item-val">${stats.nodes}</div><div class="stats-item-label">节点总数</div></div>
                        <div class="stats-item"><div class="stats-item-val">${stats.edges}</div><div class="stats-item-label">关系总数</div></div>
                        <div class="stats-item"><div class="stats-item-val">${stats.communities || 0}</div><div class="stats-item-label">社区数量</div></div>
                     </div>`;
            
            let badges = '';
            if (stats.types && Object.keys(stats.types).length > 0) {
                for (const [type, count] of Object.entries(stats.types)) { badges += `<div class="stats-badge"><div class="stats-badge-dot" style="background:${nodeColors[type]}"></div>${type}: ${count}</div>`; }
            }
            const edgeTypes = {}; fullGraphData.edges.forEach(e => { edgeTypes[e.type] = (edgeTypes[e.type] || 0) + 1; });
            for (const [type, count] of Object.entries(edgeTypes)) { badges += `<div class="stats-badge"><div class="stats-badge-line" style="background:${edgeColors[type]}"></div>${type}: ${count}</div>`; }
            if(badges) html += `<div class="stats-types">${badges}</div>`;
            
            document.getElementById('stats-container').innerHTML = html;
        } catch (e) { document.getElementById('stats-container').innerHTML = '<div class="empty-state">获取统计信息失败</div>'; }
    }

    function filterLocalData() {
        const kw = searchKeyword.toLowerCase();
        let kwNodes = fullGraphData.nodes; let kwEdges = fullGraphData.edges;
        if (kw) {
            kwNodes = kwNodes.filter(n => (n.name && n.name.toLowerCase().includes(kw)) || (n.description && n.description.toLowerCase().includes(kw)) || (n.content && n.content.toLowerCase().includes(kw)));
            const nodeMap = {}; fullGraphData.nodes.forEach(n => nodeMap[n.id] = n.name);
            kwEdges = kwEdges.filter(e => (e.type && e.type.toLowerCase().includes(kw)) || (e.instruction && e.instruction.toLowerCase().includes(kw)) || (e.condition && e.condition.toLowerCase().includes(kw)) || (nodeMap[e.from_id] && nodeMap[e.from_id].toLowerCase().includes(kw)) || (nodeMap[e.to_id] && nodeMap[e.to_id].toLowerCase().includes(kw)));
        }
        let listNodes = kwNodes; let listEdges = kwEdges;
        if (currentTab === 'nodes') { if (filterType !== 'ALL') listNodes = listNodes.filter(n => n.type === filterType); }
        else { if (filterType !== 'ALL') listEdges = listEdges.filter(e => e.type === filterType); }
        const nodeMap = {}; fullGraphData.nodes.forEach(n => nodeMap[n.id] = n.name);
        renderLists(listNodes, listEdges, nodeMap); renderTables(listNodes, listEdges, nodeMap);
        
        const validNodeIds = new Set(kwNodes.map(n => n.id)); const validEdgeIds = new Set(kwEdges.map(e => e.id));
        nodes.update(fullGraphData.nodes.map(n => { const isValid = validNodeIds.has(n.id); const baseColor = nodeColors[n.type] || '#6b7280'; return { id: n.id, color: { background: isValid ? baseColor : hexToRgba(baseColor, 0.15), border: isValid ? baseColor : hexToRgba(baseColor, 0.15), highlight: { background: '#60a5fa', border: '#60a5fa' } }, font: { color: isValid ? '#ffffff' : 'rgba(255,255,255,0.1)' } }; }));
        edges.update(fullGraphData.edges.map(e => { const isValid = validEdgeIds.has(e.id); const baseColor = edgeColors[e.type] || '#4b5563'; return { id: e.id, color: { color: isValid ? baseColor : hexToRgba(baseColor, 0.05), highlight: isValid ? '#3b82f6' : hexToRgba(baseColor, 0.05) }, font: { color: isValid ? '#a2a5b9' : 'rgba(162, 165, 185, 0.05)', strokeWidth: 0 } }; }));
    }

    async function submitAddEdge() { const from_id = document.getElementById('add-edge-from').value; const to_id = document.getElementById('add-edge-to').value; if(!from_id || !to_id) return showToast('请从下拉列表中选择节点', 'error'); if(from_id === to_id) return showToast('起始节点和目标节点不能相同', 'error'); const data = { from_id, to_id, type: document.getElementById('add-edge-type').value, instruction: document.getElementById('add-edge-inst').value }; const res = await fetch('/api/edge', { method: 'POST', headers: {'Content-Type': 'application/json'}, body: JSON.stringify(data) }); if(res.ok) { showToast('关系创建成功'); closeModal('add-edge-modal'); await loadGraph(); } else showToast('创建失败', 'error'); }
